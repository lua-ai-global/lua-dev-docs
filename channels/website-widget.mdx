## Overview

LuaPop is an embeddable chat widget that brings your AI agent directly to your website - no channel connection needed, just add a script tag!

## Quick Setup

<Steps>
  <Step title="Add Script to Website">
    ```html
    <script src="https://lua-ai-global.github.io/lua-pop/lua-pop.umd.js"></script>
    <script>
      window.LuaPop.init({
        agentId: "your-agent-id",
        position: "bottom-right",
        theme: "dark"
      });
    </script>
    ```
    
    Place before closing `</body>` tag
  </Step>
  
  <Step title="Replace Agent ID">
    Get your agent ID from:
    
    ```bash
    lua production
    ```
    
    Or from `lua.skill.yaml`
  </Step>
  
  <Step title="Refresh Page">
    Widget appears immediately!

      <img src="/images/channels/website-widget-button.png" alt="Widget button" />
    </Frame>
    
    *Screenshot: Website with chat button in bottom-right*
  </Step>
  
  <Step title="Test Chat">
    Click button and send message
    
    <Frame>
      <img src="/images/channels/website-widget-chat.png" alt="Chat window" />
    </Frame>
    
    *Screenshot: Open chat window with conversation*
  </Step>
</Steps>

## Framework Integration

<Tabs>
  <Tab title="React">
    ```tsx
    import { useEffect } from 'react';
    
    export default function ChatWidget() {
      useEffect(() => {
        const script = document.createElement('script');
        script.src = 'https://lua-ai-global.github.io/lua-pop/lua-pop.umd.js';
        script.onload = () => {
          window.LuaPop?.init({
            agentId: process.env.REACT_APP_AGENT_ID,
            position: "bottom-right"
          });
        };
        document.body.appendChild(script);
      }, []);
      return null;
    }
    ```
  </Tab>

  <Tab title="Next.js">
    ```tsx
    import Script from 'next/script';
    
    export default function Layout({ children }) {
      return (
        <>
          {children}
          <Script 
            src="https://lua-ai-global.github.io/lua-pop/lua-pop.umd.js"
            onLoad={() => {
              window.LuaPop?.init({
                agentId: process.env.NEXT_PUBLIC_AGENT_ID
              });
            }}
          />
        </>
      );
    }
    ```
  </Tab>

  <Tab title="WordPress">
    Add to theme's `footer.php` or use "Insert Headers and Footers" plugin
  </Tab>

  <Tab title="Shopify">
    Add to `theme.liquid` before `</body>`
  </Tab>
</Tabs>
